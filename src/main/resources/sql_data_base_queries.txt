create database shop;

create table users(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	username varchar(50),
	email varchar(50),
	password varchar(50),
	walletId varchar(50),
	role varchar(50),
	active boolean
);

create table tokens(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	userId varchar(50),
	token varchar(50),
	tokenType varchar(50),
	email varchar(50),
	role varchar(50),
	usageAmount int
);

create table products(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	ownerShopId varchar(50),
	name varchar(50),
	amount int,
	price real
);

create table orders(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	productName varchar(50),
	amount int,
	price real,
	merchantId varchar(50),
	customerId varchar(50)
);

create table shops(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	name varchar(50),
	ownerId varchar(50),
	walletId varchar(50)
);

create table temp_codes(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	email varchar(50),
	tempCode int
);

create table wallets(
	id VARCHAR(50) primary key not null,
	createdDate DATE,
	lastUpdatedDate DATE,
	ownerId varchar(50),
	ownerType varchar(50),
	amount real
);
